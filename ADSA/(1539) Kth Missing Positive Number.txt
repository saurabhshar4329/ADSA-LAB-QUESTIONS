class Solution {
    public int findKthPositive(int[] arr, int k) {
        int left = 0, right = arr.length - 1;

        while (left <= right) {
            int mididx = left + (right - left) / 2;
            int missing = arr[mididx] - (mididx + 1);

            if (missing < k) {
                left = mididx + 1;
            } else {
                right = mididx - 1;
            }
        }

        return left + k;
    }
}
